package com.picchealth.module.sys.api;

import com.picchealth.config.ApiInterceptor;
import com.picchealth.module.system.dto.ExportPermissionDto;

/**
 * 权限归属接口

    private PersonDivisionService personDivisionService;

### 5. 创建 Service 实现

**文件路径：** `picc-mzmtb-user/picchealth-privilege-server/src/main/java/com/picchealth/module/sys/service/impl/PersonDivisionServiceImpl.java`

```java
package com.picchealth.module.sys.service.impl;

import com.picchealth.comm.service.impl.BaseServiceImpl;
import com.picchealth.module.system.dao.PersonDivisionDao;
import com.picchealth.module.system.dto.ExportPermissionDto;
import com.picchealth.module.system.po.PersonDivision;
import com.picchealth.module.sys.service.PersonDivisionService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import javax.annotation.Resource;
import java.util.List;

@Slf4j
@Service("personDivisionService")
public class PersonDivisionServiceImpl extends BaseServiceImpl<PersonDivision> implements PersonDivisionService {

    @Resource
    private PersonDivisionDao personDivisionDao;

    @Override
    public ExportPermissionDto getExportPermission(String account, String flag) {
        ExportPermissionDto result = new ExportPermissionDto();
        result.setAccount(account);
        result.setFlag(flag);
        result.setIsexport("0"); // 默认无权限
        
        try {
            // 查询权限归属
            PersonDivision query = new PersonDivision();
            query.setAccount(account);
            query.setFlag(flag);
            query.setDr((short) 0); // 有效数据
            
            List<PersonDivision> list = personDivisionDao.queryPersonDivisonList(query);
            
            if (list != null && !list.isEmpty()) {
                PersonDivision pd = list.get(0);
                // 判断是否有导出权限
                if ("1".equals(pd.getIsexport())) {
                    result.setIsexport("1");
                    log.info("账号 {} 在地市 {} 有导出权限", account, flag);
                } else {
                    log.info("账号 {} 在地市 {} 无导出权限", account, flag);
                }
            } else {
                log.warn("账号 {} 在地市 {} 无权限归属配置", account, flag);
            }
        } catch (Exception e) {
            log.error("查询导出权限失败: account={}, flag={}", account, flag, e);
        }
        
        return result;
    }
}
```

### 6. 创建数据脱敏工具类

**文件路径：** `picc-mzmtb-server/picchealth-server/src/main/java/com/picchealth/utils/SensitiveDataUtil.java`

```java
package com.picchealth.utils;

import org.apache.commons.lang3.StringUtils;

/**
 * 敏感数据脱敏工具类
 */
public class SensitiveDataUtil {

    /**
     * 姓名脱敏
     * 张三 -> 张**
     * 李小明 -> 李**
     */
    public static String maskName(String name) {
        if (StringUtils.isBlank(name)) {
            return "";
        }
        if (name.length() == 1) {
            return name + "**";
        }
        return name.substring(0, 1) + "**";
    }

    /**
     * 身份证号脱敏
     * 130180199802180918 -> 130180********0918
     */
    public static String maskIdCard(String idCard) {
        if (StringUtils.isBlank(idCard) || idCard.length() < 10) {
            return idCard;
        }
        return idCard.substring(0, 6) + "********" + idCard.substring(idCard.length() - 4);
    }

    /**
     * 手机号脱敏
     * 15409287181 -> 154****7181
     */
    public static String maskPhone(String phone) {
        if (StringUtils.isBlank(phone) || phone.length() < 7) {
            return phone;
        }
        return phone.substring(0, 3) + "****" + phone.substring(phone.length() - 4);
    }
}
```

### 7. 修改榆林导出 Service

**文件路径：** `picc-mzmtb-server/mtb-yl/src/main/java/com/picchealth/module/mtb/service/impl/YIExcelServiceImpl.java`

在导出方法中添加权限判断逻辑：

```java
package com.picchealth.module.mtb.service.impl;

import com.picchealth.client.UserClient;
import com.picchealth.module.mb.dto.ExportPermissionDto;
import com.picchealth.utils.SensitiveDataUtil;
import com.picchealth.utils.UserUtils;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import pdfc.framework.web.ApiResponse;

import javax.annotation.Resource;

@Slf4j
@Service
public class YIExcelServiceImpl {

    @Resource
    private UserClient userClient;  // Feign 客户端

    /**
     * 判断当前用户是否有导出敏感信息权限
     * @return true: 有权限（不脱敏），false: 无权限（需脱敏）
     */
    private boolean hasExportPermission() {
        try {
            // 获取当前登录用户账号
            String account = UserUtils.getUser().getAccount();
            String flag = "15"; // 榆林地市标识
            
            // 调用 user 服务获取权限
            ExportPermissionDto dto = new ExportPermissionDto();
            dto.setAccount(account);
            dto.setFlag(flag);
            
            ApiResponse<ExportPermissionDto> response = userClient.getExportPermission(dto);
            
            if (response != null && response.getStatus() == 0 && response.getData() != null) {
                return "1".equals(response.getData().getIsexport());
            }
        } catch (Exception e) {
            log.error("获取导出权限失败", e);
        }
        
        // 默认无权限
        return false;
    }

    /**
     * 导出申报详情（带权限控制）
     */
    public void exportDeclareDetail(List<VipMbdeclareInfoDto> dataList) {
        // 判断是否有导出权限
        boolean hasPermission = hasExportPermission();
        
        log.info("导出敏感信息权限: {}", hasPermission ? "有权限（不脱敏）" : "无权限（需脱敏）");
        
        // 遍历数据进行脱敏处理（如果无权限）
        if (!hasPermission) {
            for (VipMbdeclareInfoDto data : dataList) {
                // 参保人姓名脱敏
                data.setName(SensitiveDataUtil.maskName(data.getName()));
                
                // 参保人手机号脱敏
                data.setMobile(SensitiveDataUtil.maskPhone(data.getMobile()));
                
                // 参保人身份证号脱敏
                data.setIdcard(SensitiveDataUtil.maskIdCard(data.getIdcard()));
                
                // 专家姓名脱敏
                if (data.getVipMbdeclareApprovalDtos() != null) {
                    for (VipMbdeclareApprovalDto approval : data.getVipMbdeclareApprovalDtos()) {
                        if (approval.getApprovalusername() != null) {
                            approval.setApprovalusername(SensitiveDataUtil.maskName(approval.getApprovalusername()));
                        }
                        // 专家手机号脱敏
                        if (approval.getApprovalphone() != null) {
                            approval.setApprovalphone(SensitiveDataUtil.maskPhone(approval.getApprovalphone()));
                        }
                    }
                }
            }
        }
        
        // 执行导出逻辑
        // ... 原有导出代码 ...
    }
}
```
