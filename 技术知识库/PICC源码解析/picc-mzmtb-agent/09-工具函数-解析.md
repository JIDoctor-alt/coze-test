#### 1.1 AES 加密/解密
```javascript
// 加密
let encrypt = (word) => {
    let aseNum = process.env.ASE_Num;
    var key = CryptoJS.enc.Utf8.parse(aseNum);
    var srcs = CryptoJS.enc.Utf8.parse(word);
    var encrypted = CryptoJS.AES.encrypt(srcs, key, {
        mode: CryptoJS.mode.ECB,
        padding: CryptoJS.pad.Pkcs7
    });
    return encrypted.toString();
}

// 解密
let decrypt = (word) => {
    try {
        let aseNum = process.env.ASE_Num;
        var key = CryptoJS.enc.Utf8.parse(aseNum);
        var decrypt = CryptoJS.decrypt(word, key, {
            mode: CryptoJS.mode.ECB,
            padding: CryptoJS.pad.Pkcs7
        });
        return CryptoJS.enc.Utf8.stringify(decrypt).toString();
    } catch (error) {
        console.log(error);
    }
}
```

#### 1.2 SM4 加密/解密
```javascript
// SM4加密（国密算法）
let sm4Encrypt = async (word, params = {}) => {
    const res = await api.encryptionSm4(params);
    if (res.status == "0" && res.data) {
        const pwdKey = decrypt(res.data);  // 解密获取密钥
        let sm4Config = {
            key: pwdKey,
            mode: "ecb",
            cipherType: "base64",
            padding: 'pkcs7',
        };
        return new SM4(sm4Config).encrypt(word);
    }
}

// SM4解密
let sm4Decrypt = async (word, params = {}) => {
    const res = await api.encryptionSm4(params);
    if (res.status == "0" && res.data) {
        const pwdKey = decrypt(res.data);
        let sm4Config = {
            key: pwdKey,
            mode: "ecb",
            cipherType: "base64",
            padding: 'pkcs7',
        };
        return new SM4(sm4Config).decrypt(word);
    }
}
```

    params = new Date(params);
    var year = params.getFullYear();
    var month = params.getMonth() + 1;
    var date = params.getDate();

// 补零
let addZero = (params) => {
    return params.toString().length > 1 ? params.toString() : '0' + params.toString();

let formatDate = (n) => {

## 2. tableMixins.js - 表格混入

### 位置
`utils/tableMixins.js`

### 核心功能
封装表格的通用逻辑，减少代码重复。

```javascript
export default {
    data() {
        return {
            // 默认分页配置
            pagination: {
                current: 1,
                pageSize: 10,
                total: 0,
                showTotal: total => `共 ${total} 条`,
                showSizeChanger: true,
                pageSizeOptions: ["10", "20", "35", "50"]
            }
        }
    },
    methods: {
        // 切换每页条数
        onPageSizeChange(current, pageSize) {
            this.pagination.pageSize = pageSize;
            this.pagination.current = 1;
            this.initPage && this.initPage();
        },
        
        // 切换页码
        onPageChange(page, pageSize) {
            this.pagination.current = page;
            this.initPage && this.initPage();
        },
        
        // 重置分页
        resetPagination() {
            this.pagination.current = 1;
        }
    }
}
```

### 使用方式
```javascript
import tableMixins from "@/utils/tableMixins";

export default {
    mixins: [tableMixins],
    // 自动获得 pagination 和分页方法
    // 只需实现 initPage 方法即可
}
```

```javascript
// Base64格式文件下载
export const downloadFileByBase64 = (base64Data, fileName) => {
    const link = document.createElement('a');
    link.href = 'data:application/vnd.ms-excel;base64,' + base64Data;
    link.download = fileName || '下载文件.xlsx';
    link.click();
}

// Blob格式文件下载
export const downloadFileByBlob = (blob, fileName) => {
    const url = window.URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.download = fileName;
    link.click();
    window.URL.revokeObjectURL(url);
}
```

### 🔹 加密算法对比
| 算法 | 说明 | 用途 |
|------|------|------|
| MD5 | 不可逆哈希 | 密码存储（已不建议单独使用） |
| AES | 对称加密 | 数据加密传输 |
| SM4 | 国密对称加密 | 国产化要求的数据加密 |

### 🔹 浮点数精度问题
```javascript
// JavaScript 浮点数运算不精确
0.1 + 0.2 = 0.30000000000000004  // ❌

// 解决方案：转换为整数运算
(0.1 * 10 + 0.2 * 10) / 10 = 0.3  // ✅
```
